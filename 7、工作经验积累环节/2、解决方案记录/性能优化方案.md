## 为了提高性能跑分，在抛开编译优化，使用不同编译器和编译工具的情况下，这里记录将coremark程序存放到SRAM中运行，以提高跑分结果

1、在coremark.h文件中定义了一个宏
#define RAM_FUNCTION __attribute__((section(".coremark")))
然后在coremark.c文件中，将coremark函数添加该宏
如果不需要该宏的时候，可以将这个宏定义为空
#define RAM_FUNCTION

2、在链接脚本中设置段 
```C
 .coremark_section :
{
	. = ALIGN(4);
	__coremark_start = .;
	KEEP(*(.coremark_section))
	__coremark_end = .;
	. = ALIGN(4);
} > m_diyu AT > m_text

__coremark_loadaddr = LOADADDR(.coremark_section);
__coremark_size = __coremark_end - __coremark_start;
```
3、然后加上一段搬运代码，将代码从下载的位置搬运到 RAM 中

```c
/* Reset Handler */

    .thumb_func         //申明这是一个Thumb函数（cortex-M只支持Thumb模式）
    .align 2            //对齐到4字节边界
    .globl   Reset_Handler  //申明这是一个全局函数,对外可见
    .weak    Reset_Handler  //声明为弱符号，允许用户重新实现
    .type    Reset_Handler, %function   //告诉汇编器 这是一个函数
Reset_Handler:              //函数入口
    cpsid   i               /* Mask interrupts */
                            //关闭全局中断（CPS = change progressor state, ID = interrupt disable）
#ifndef __NO_SYSTEM_INIT
    bl SystemInit           //调用系统初始化函数（配置时钟、PLL等）
#endif
/*    bl init_data_bss  jpan*/
    cpsie   i               /* Unmask interrupts *///开启全局中断

/* 为什么要关中断呢，是因为初始化过程，比如复制数据、清零BSS段 不希望被中断打断 */
/* 防止在环境准备好之前发生中断 */

/*
// SystemInit 的作用 
1、配置系统时钟，外设时钟
2、设置Flash等待周期
3、初始化其他系统级外设
*/


// 加载链接脚本中定义的符号地址
/* from m3 project  */
    ldr    r1, =__etext             // R1 = Flash中data段的起始地址（初始值存储位置）
    ldr    r2, =__data_start__      // R2 = RAM中data段的起始地址
    ldr    r3, =__data_end__        // R3 = RAM中data段的结束地址

    subs    r3, r2                  // R3 = 数据长度（结束-开始）
    ble    .LC1                     // 如果长度 <=0 ,跳过复制，
.LC0:                               // 复制循环开始
    subs    r3, #4                  // 长度减4，每次复制4个字节
    ldr    r0, [r1, r3]             // 从Flash中加载4个字节到R0
    str    r0, [r2, r3]             // 存储4个字节到RAM中
    bgt    .LC0                     //如果还有数据，继续循环
.LC1:
    ldr     r1, = __coremark_loadaddr   // R1 = Flash中coremark代码的起始地址
    ldr     r2, = __coremark_start__  // R2 = RAM中coremark代码的起始地址
    ldr     r3, = __coremark_end__    // R3 = RAM中coremark代码的结束地址

    cmp     r2, r3                      // 如果RAM中coremark代码的起始地址和结束地址相等，说明coremark代码没有被搬运到RAM中
    beq    .LC3                          // 如果相等，说明coremark代码没有被搬运到RAM中，直接跳转到coremark代码的起始地址

    subs    r0, r3,r2                   // R0 = coremark代码的长度（结束-开始）   
.LC2    
    cmp     r0, #0                     //检查是否完成复制 
    beq     .LC3                        //完成则跳出
    subs    r0, #4                      //长度减4
    ldr     r4, [r1, r0]                // 从Flash中加载4个字节到R4
    str     r4, [r2, r0]                // 存储4个字节到RAM中   
    bgt     .LC2
.LC3:
    dsb
    isb

#ifdef __STARTUP_CLEAR_BSS
/*     This part of work usually is done in C library startup code. Otherwise,
 *     define this macro to enable it in this startup.
 *
 *     Loop to zero out BSS section, which uses following symbols
 *     in linker script:
 *      __bss_start__: start of BSS section. Must align to 4
 *      __bss_end__: end of BSS section. Must align to 4
 */
    ldr r1, =__bss_start__
    ldr r2, =__bss_end__

    movs    r0, 0
.LC4:
    cmp     r1, r2
    itt    lt
    strlt   r0, [r1], #4
    blt    .LC4
#endif /* __STARTUP_CLEAR_BSS */
```


## 这里记录一些汇编语法 

- ldr     r4, [r1, r0]                              // 意思是去内存地址 r1 + r0 这个地址，把那里的四个字节取出来，放到 r4 寄存器中
- │       │   │    │
- │       │   │    └── 偏移量寄存器 (r0)
- │       │   └─────── 基址寄存器 (r1)
- │       └─────────── 目标寄存器 (r4)
- └─────────────────── 指令：加载32位字
- 
- 执行结果 ： 
- 计算地址：r1 + r0
- 从该地址读取4个字节（32位）
- 将这4个字节存入r4寄存器

- str     r4, [r2, r0]                              // 意思是 把 r4 寄存器中的4个字节写入到内存地址 r2 + r0 这个地址
- │       │   │    │
- │       │   │    └── 偏移量寄存器 (r0)
- │       │   └─────── 基址寄存器 (r2)
- │       └─────────── 源寄存器 (r4)
- └─────────────────── 指令：存储32位字
- 
- 执行结果：
- 计算地址：r2 + r0
- 将r4寄存器的4个字节写入该地址

``` asm

.LC2:
    cmp     r0, #0          ; 检查剩余字节数
    beq     .LC3            ; 如果为0，跳转结束
    subs    r0, #4          ; r0 = r0 - 4 (准备处理下4个字节)
    
    ; 关键的两条指令
    ldr     r4, [r1, r0]    ; 从Flash读取4字节：地址 = r1 + r0
    str     r4, [r2, r0]    ; 写入到RAM：地址 = r2 + r0
    
    bgt     .LC2            ; 如果r0>0，继续循环

``` 

- 执行示例：
- r1 = 0x08001000 (Flash 中Coremark 的起始地址)
- r2 = 0x20000000 (RAM 中Coremark 的目标地址)
- r3 = 0x08000000 + 0x00000200 (Flash 中Coremark 的结束地址)
- 
- r0 = r3 - r2 = 0x00000200 (Coremark 的长度)
-
